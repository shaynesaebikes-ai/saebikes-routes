<script>

var ROUTES=[];

var POI_TYPES=[{val:"winery",label:"Winery"},{val:"restaurant",label:"Restaurant"},{val:"cafe",label:"Cafe"},{val:"brewery",label:"Brewery"},{val:"poi",label:"Landmark"}];

var DIFF_LABELS={easy:"Easy","easy-moderate":"Easy / Moderate",moderate:"Moderate",hard:"Hard"};

var COLORS=["#B85C38","#5A7A47","#405B30","#2E6B9E","#8E44AD","#D35400","#1A7A6E","#7B2D2D"];

var openEditId=null,wizData=null,selColor="#B85C38",selDiff="easy-moderate",dragSrc=null;


async function loadRoutes(){

  try{

    var r=await fetch('/routes.json?t='+Date.now());

    if(!r.ok)throw 0;

    ROUTES=await r.json();

  }catch(e){ROUTES=[];}

  renderRoutes();

}


function saveRoutesJson(){

  if(openEditId!==null){

    var r=ROUTES.find(function(x){return x.id===openEditId;});

    if(r)flushEdit(r);

  }

  var blob=new Blob([JSON.stringify(ROUTES,null,2)],{type:'application/json'});

  var url=URL.createObjectURL(blob);

  var a=document.createElement('a');

  a.href=url;

  a.download='routes.json';

  document.body.appendChild(a);

  a.click();

  document.body.removeChild(a);

  URL.revokeObjectURL(url);

  showStatus('routes.json downloaded — drag it into Netlify to deploy.','info');

}


function showStatus(msg,type){

  var el=document.getElementById('stbar');

  el.textContent=msg;

  el.className='stbar show '+(type||'info');

  clearTimeout(el._t);

  if(msg)el._t=setTimeout(function(){el.classList.remove('show');},6000);

}


function esc(s){

  return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');

}


function renderRoutes(){

  var c=document.getElementById('arl');

  if(!ROUTES.length){c.innerHTML='<div class="empty">No routes yet. Drop a GPX file below to add one.</div>';return;}

  c.innerHTML='';

  ROUTES.forEach(function(r,idx){

    var row=document.createElement('div');

    row.className='arow';row.draggable=true;row.dataset.idx=idx;

    var dk=r.difficulty||'easy';

    var vp=(r.pois||[]).filter(function(p){return p.visible!==false;}).length;

    var io=(openEditId===r.id);

    row.innerHTML=

      '<div class="dhdl">&#8903;</div>'+

      '<div class="rcbar" style="background:'+r.color+'"></div>'+

      '<div class="arinfo">'+

        '<div class="arname">'+esc(r.name)+'<span class="dchip d-'+dk+'">'+DIFF_LABELS[dk]+'</span></div>'+

        '<div class="armeta">'+r.distance+' km &middot; '+r.estTime+' &middot; '+r.climb+'m ascent &middot; '+vp+' visible stops</div>'+

        buildEP(r,idx,io)+

      '</div>'+

      '<div class="aracts">'+

        '<button class="btn" onclick="toggleEdit('+r.id+')">Edit</button>'+

        '<button class="btn btn-red" onclick="deleteRoute('+r.id+')">Delete</button>'+

      '</div>';

    row.addEventListener('dragstart',function(e){dragSrc=idx;e.dataTransfer.effectAllowed='move';});

    row.addEventListener('dragover',function(e){e.preventDefault();row.classList.add('drag-over');});

    row.addEventListener('dragleave',function(){row.classList.remove('drag-over');});

    row.addEventListener('drop',function(e){

      e.preventDefault();row.classList.remove('drag-over');

      if(dragSrc===null||dragSrc===idx)return;

      var m=ROUTES.splice(dragSrc,1)[0];ROUTES.splice(idx,0,m);

      dragSrc=null;renderRoutes();

    });

    c.appendChild(row);

  });

  if(openEditId!==null){

    var r=ROUTES.find(function(x){return x.id===openEditId;});

    if(r)wireEP(r);

  }

}


function buildEP(r,idx,io){

  var h='<div class="epanel'+(io?' open':'')+'" id="ep-'+r.id+'">';

  h+='<div class="erow"><label>Route name</label><input id="en-'+r.id+'" value="'+esc(r.name)+'"></div>';

  h+='<div class="erow"><label>Description</label><textarea id="ed-'+r.id+'">'+esc(r.description)+'</textarea></div>';

  h+='<div class="erow"><label>Difficulty</label><div class="brow">';

  ['easy','easy-moderate','moderate','hard'].forEach(function(d){

    h+='<button class="obtn'+(r.difficulty===d?' sel':'')+'" data-rid="'+r.id+'" data-diff="'+d+'">'+DIFF_LABELS[d]+'</button>';

  });

  h+='</div></div>';

  h+='<div class="erow"><label>Colour</label><div class="csw">';

  COLORS.forEach(function(c){

    h+='<div class="cswatch'+(r.color===c?' sel':'')+'" data-rid="'+r.id+'" data-color="'+c+'" style="background:'+c+'"></div>';

  });

  h+='</div></div>';

  h+='<div class="erow"><div class="poi-ed-hdr"><label>Points of interest</label>'+

    '<div class="pvis-all-btns">'+

      '<button class="pvis-all-btn" data-allri="'+idx+'" data-show="1">Show all</button>'+

      '<button class="pvis-all-btn" data-allri="'+idx+'" data-show="0">Hide all</button>'+

    '</div></div><div class="poi-ed-list">';

  (r.pois||[]).forEach(function(p,pi){

    var km=p.km===0?'Start':p.km+' km';

    var vis=p.visible!==false;

    h+='<div class="poi-ed-item"><div class="poi-ed-top">'+

      '<div class="poi-ed-name">'+esc(p.name)+'</div>'+

      '<div class="poi-ed-km">'+km+'</div>'+

      '<div class="poi-vis-toggle"><label class="toggle">'+

        '<input type="checkbox"'+(vis?' checked':'')+' data-ri="'+idx+'" data-pi="'+pi+'">'+

        '<span class="tslider"></span></label>'+

        '<span class="vis-lbl">'+(vis?'Shown':'Hidden')+'</span>'+

      '</div></div><div class="poi-ed-types">';

    POI_TYPES.forEach(function(t){

      h+='<button class="ptbtn'+(p.type===t.val?' sel':'')+'" data-ri="'+idx+'" data-pi="'+pi+'" data-type="'+t.val+'">'+t.label+'</button>';

    });

    h+='</div><div class="poi-ed-desc"><input id="pdesc-'+r.id+'-'+pi+'" value="'+esc(p.desc||'')+'" placeholder="POI description..."></div></div>';

  });

  h+='</div></div></div>';

  return h;

}


function wireEP(r){

  document.querySelectorAll('[data-diff][data-rid="'+r.id+'"]').forEach(function(b){

    b.onclick=function(){r.difficulty=b.dataset.diff;renderRoutes();};

  });

  document.querySelectorAll('[data-color][data-rid="'+r.id+'"]').forEach(function(s){

    s.onclick=function(){r.color=s.dataset.color;renderRoutes();};

  });

  document.querySelectorAll('input[data-ri][data-pi]').forEach(function(cb){

    cb.onchange=function(){

      var ri=parseInt(cb.dataset.ri),pi=parseInt(cb.dataset.pi);

      ROUTES[ri].pois[pi].visible=cb.checked;

      cb.closest('.poi-vis-toggle').querySelector('.vis-lbl').textContent=cb.checked?'Shown':'Hidden';

    };

  });

  document.querySelectorAll('.ptbtn[data-ri][data-pi]').forEach(function(b){

    b.onclick=function(){

      var ri=parseInt(b.dataset.ri),pi=parseInt(b.dataset.pi);

      ROUTES[ri].pois[pi].type=b.dataset.type;

      b.closest('.poi-ed-types').querySelectorAll('.ptbtn').forEach(function(x){x.classList.remove('sel');});

      b.classList.add('sel');

    };

  });

  document.querySelectorAll('[data-allri]').forEach(function(b){

    b.onclick=function(){

      var ri=parseInt(b.dataset.allri),show=b.dataset.show==='1';

      ROUTES[ri].pois.forEach(function(p){p.visible=show;});

      renderRoutes();

    };

  });

}


function flushEdit(r){

  var n=document.getElementById('en-'+r.id),d=document.getElementById('ed-'+r.id);

  if(n)r.name=n.value.trim()||r.name;

  if(d)r.description=d.value.trim();

  (r.pois||[]).forEach(function(p,pi){

    var el=document.getElementById('pdesc-'+r.id+'-'+pi);

    if(el)p.desc=el.value.trim();

  });

}


function toggleEdit(id){

  if(openEditId!==null&&openEditId!==id){

    var p=ROUTES.find(function(x){return x.id===openEditId;});

    if(p)flushEdit(p);

  }

  openEditId=(openEditId===id)?null:id;

  renderRoutes();

}


function deleteRoute(id){

  if(!confirm('Delete this route?'))return;

  ROUTES=ROUTES.filter(function(r){return r.id!==id;});

  if(openEditId===id)openEditId=null;

  renderRoutes();

}


function handleGpx(file){

  var reader=new FileReader();

  reader.onload=function(e){

    try{

      var xml=new DOMParser().parseFromString(e.target.result,'text/xml');

      var trkpts=xml.querySelectorAll('trkpt');

      if(!trkpts.length){showStatus('No track points found in GPX.','error');return;}

      var coords=[],elKm=[],elM=[],dist=0,prev=null;

      trkpts.forEach(function(pt){

        var lat=parseFloat(pt.getAttribute('lat'));

        var lng=parseFloat(pt.getAttribute('lon'));

        var ele=parseFloat((pt.querySelector('ele')||{}).textContent||'0');

        if(prev)dist+=Math.hypot(lat-prev[0],lng-prev[1])*111.32;

        coords.push([+lat.toFixed(5),+lng.toFixed(5)]);

        elKm.push(+dist.toFixed(2));

        elM.push(+ele.toFixed(1));

        prev=[lat,lng];

      });

      var climb=0;

      for(var i=1;i<elM.length;i++){var d=elM[i]-elM[i-1];if(d>0)climb+=d;}

      var wpts=xml.querySelectorAll('wpt');

      var pois=[{name:'Start',type:'poi',desc:'',lat:coords[0][0],lng:coords[0][1],km:0,visible:true}];

      wpts.forEach(function(wpt){

        var wlat=parseFloat(wpt.getAttribute('lat'));

        var wlng=parseFloat(wpt.getAttribute('lon'));

        var wname=(wpt.querySelector('name')||{}).textContent||'Stop';

        var wdesc=(wpt.querySelector('desc')||{}).textContent||'';

        var nearest=0,nearDist=Infinity;

        coords.forEach(function(c,i){var dd=Math.hypot(c[0]-wlat,c[1]-wlng);if(dd<nearDist){nearDist=dd;nearest=i;}});

        pois.push({name:wname.trim(),type:'poi',desc:wdesc.trim(),lat:+wlat.toFixed(6),lng:+wlng.toFixed(6),km:+(elKm[nearest]||0).toFixed(1),visible:false});

      });

      pois.sort(function(a,b){return a.km-b.km;});

      var em=Math.round(dist/15*60);

      wizData={coords:coords,elevKm:elKm,elevM:elM,distance:+dist.toFixed(1),climb:Math.round(climb),estTime:Math.floor(em/60)+'h '+(em%60)+'m',pois:pois,filename:file.name};

      selColor=COLORS[ROUTES.length%COLORS.length];

      selDiff='easy-moderate';

      renderWizard();

    }catch(err){showStatus('Error parsing GPX: '+err.message,'error');}

  };

  reader.readAsText(file);

}

function selAllWizPois(show){

  if(!wizData)return;

  wizData.pois.forEach(function(p,i){

    p.visible=show;

    var cb=document.querySelector('[data-wpi-vis="'+i+'"]');if(cb)cb.checked=show;

    var lbl=document.getElementById('wvl-'+i);if(lbl)lbl.textContent=show?'Shown':'Hidden';

  });

}


function addRoute(){

  var name=(document.getElementById('wname').value||'').trim();

  var desc=(document.getElementById('wdesc').value||'').trim();

  if(!name){showStatus('Please enter a route name.','warn');return;}

  var maxId=ROUTES.reduce(function(m,r){return Math.max(m,r.id||0);},0);

  ROUTES.push({

    id:maxId+1,name:name,

    tagline:desc.split('.')[0]||name,

    difficulty:selDiff,

    distance:wizData.distance,climb:wizData.climb,estTime:wizData.estTime,

    description:desc,color:selColor,

    coords:wizData.coords,elevKm:wizData.elevKm,elevM:wizData.elevM,

    pois:wizData.pois

  });

  wizData=null;

  document.getElementById('wiz').className='wizard';

  document.getElementById('wiz').innerHTML='';

  renderRoutes();

  showStatus('Route "'+name+'" added. Click Download routes.json to save.','info');

}
